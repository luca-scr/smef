package="smef"

# compila e installa 
Rlib="/Library/Frameworks/R.framework/Rlib"
cd ~/R

# crea package
R CMD build --compact-vignettes --force $package

# filename of latest package version
pkgfile=$(ls -t -1 $package*.tar.gz | head -1)

# remove old package
R CMD REMOVE --library="$Rlib" $package

# install package
R CMD INSTALL --library="$Rlib" $pkgfile
